    抽象类

    例如要写一个动物类有一个行走方法，但是每个动物子类的行走方式都不一样,
    所以动物类要抽象化实现，就是不写方法 不能用abstract修饰属性，私有方法，构造器，静态方法，final方法
    对象都是通过类来描绘的，
    但是反过来，并不是所有的类都是用来描绘对象的，
    如果一个类中没有包含足够的信息来描绘一个具体的对象，这样的类就是抽象类
    抽象类除了不能实例化对象之外，类的其它功能依然存在，
    成员变量、成员方法和构造方法的访问方式和普通类一样
    因为不能实例化所以要继承才能用
    在类中声明一个方法，这个方法没有实现体，是一个“空”方法    		 
    public abstract class Hero {
    String name;
    float hp;

    练习：抽象Employee类

    abstract public class Employee { // 抽象化一个员工类
    public Employee() {             //程序中如果定义了有参构造方法，就不会生成默认的空构造方法，所以一                                     //般如果没有定义构造方法要先定义一个（）是空{}里可以输出

    }
    // 有姓名工号工资属性
    String name;
    int id;
    double salary;

    public abstract void work(); // 一个空方法
    }

class CommonEmployee extends Employee {
    @Override
    public void work() { // 继承employee后的work方法然后给一个方法体
    // TODO Auto-generated method stub
        System.out.println("这是一个普通员工");
    }

    public void setCommonEmployeeinfo(String name, int id, double salary) { // 访问父类的的    属性用super
        super.name = name;
        super.id = id;
        super.salary = salary;
    }

    public void getCommonEmplloyinfo() { // 输出属性
        System.out.println(super.name);
        System.out.println(super.id);
        System.out.println(super.salary);
    }
}

class Manager extends Employee {
    double bonus;
public void setManagerinfo(String name, int id, double salary, double bonus) {
    super.name = name;
    super.id = id;
    super.salary = salary;
    this.bonus = bonus;
}

public void getManagerinfo() {
    System.out.println(super.name);
    System.out.println(super.id);
    System.out.println(super.salary);
    System.out.println(this.bonus); // 用this访问自己的属性，其实用super也可以因为继承了就是自己的
}

@Override
public void work() {
    // TODO Auto-generated method stub
    System.out.println("这是一个领导");
}    
}
